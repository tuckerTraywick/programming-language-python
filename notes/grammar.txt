program
	= ?namespaceStatement *programStatement

namespaceStatement
	= "namspace" qualifiedName

programStatement
	= usingStatement
	| variableDefinition
	| functionDefinition
	| structDefinition
	| traitDefinition

usingStatement
	= "using" qualifiedName *("," ?"." qualifiedName)

qualifiedName
	= identifier *("." identifier)

variableDefinition
	= ?"pub" "var" identifier (type ?assignment | assignment)

functionDefinition
	= ?"pub" "func" functionParameters ?type block

assignment
	= "=" expression

structDefinition
	= ?"pub" "struct" identifier ?genericParameters structBody

structBody
	= "{" *((structField | usingType) ";") "}"

structField
	= ?"pub" identifier type

usingType
	= "using" type *("," type)

type
	= tupleType
	| arrayType
	| pointerType
	| functionType
	| mutableType
	| basicType

tupleType
	= "(" *(type ",") *type ")"

arrayType
	= "[" ?expression "]" ?type

pointerType
	= "&" ?type

functionType
	= "func" ?functionParameters ?type

functionParameters
	= "(" *(functionParameter ",") ?functionParameter ")"

functionParameter
	= identifier type

mutableType
	= "mut" type

basicType
	= identifier *("." identifier) *genericArguments

genericArguments
	= "generic <" *(genericArgument ",") ?genericArgument ">"

genericArgument
	= type
	| expression
