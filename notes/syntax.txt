pub namespace animals
using std.io.printLine

pub trait Animal {
	pub name string
	pub age uint32
}

pub method makeNoise(animal &Animal)

pub struct Dog {
	using Animal
	pub tag string
}

pub struct Cat {
	using Animal
	pub breed string
}

pub func makeNoise(dog &Dog) {
	printLine("woof")
}

pub func makeNoise(cat &Cat) {
	printLine("meow")
}

pub func main() {
	var pet mut &Animal = &Dog()
	makeNoise(pet)
	pet = &Cat()
	pet->makeNoise()
}
